```CPP
#include <iostream>
using namespace std;

class Complex
{
public:
	float realPart;
	float imaginaryPart;

	//Constructor，默认值实部虚部皆为0
	Complex(float _real=0, float _imaginary=0)
	{
		realPart = _real;
		imaginaryPart = _imaginary;
	}
};

//重载<<运算符
ostream& operator<<(ostream& _cout, Complex& _complex)
{
	cout << "(" << _complex.realPart << " + " << _complex.imaginaryPart << " i)";
	
	//返回cout自身，保证cout后的<<可以链式调用
	return cout;
}
//重载>>运算符
istream& operator>>(istream& _cin, Complex& _complex)
{
	cout << "enter the realPart of the Complex number" << endl;
	cin >> _complex.realPart;
	cout << "enter the imaginaryPart of the Complex number" << endl;
	cin >> _complex.imaginaryPart;
	return cin;
}
//重载*运算符
Complex& operator*(Complex& c1, Complex& c2)
{
	Complex result;
	result.realPart = (c1.realPart * c2.realPart) - (c1.imaginaryPart * c2.imaginaryPart);
	result.imaginaryPart = (c1.realPart * c2.imaginaryPart) + (c1.imaginaryPart * c2.realPart);
	return result;
}
//重载==运算符
bool operator==(Complex& c1, Complex& c2)
{
	//实部虚部都相等才能判定此复数相等
	if ((c1.realPart == c2.realPart) && (c1.imaginaryPart == c2.imaginaryPart))
	{
		return true;
	}
	else
	{
		return false;
	}
}
//重载!=运算符
bool operator!=(Complex& c1, Complex& c2)
{
	//实部虚部只要有一个不相等，复数都是不相等的
	if ((c1.realPart == c2.realPart) && (c1.imaginaryPart == c2.imaginaryPart))
	{
		//这里正好与==的判定相反即可
		return false;
	}
	else
	{
		return true;
	}
}

//测试<<运算符与>>运算符的重载
void TestIO()
{
	Complex c;
	cin >> c;
	cout << "the Complex equals " << endl << c << endl;
	cout << "---------------------------------------------------------------------------" << endl;
}
//测试*运算符的重载
void TestMultiply()
{
	Complex c_1(2, 3);
	Complex c_2(4, 5);
	Complex	c = c_1 * c_2;
	//展示乘法结果
	cout << c_1 << " * " << c_2 << " = " << c << endl;
	cout << "---------------------------------------------------------------------------" << endl;
}
//测试==与!=运算符的重载
void TestEqual()
{
	Complex c_1(2, 3);
	Complex c_2(4, 5);
	Complex c_3(4, 5);
	bool test_1 = c_1 == c_2;
	bool test_2 = c_2 == c_3;
	bool test_3 = c_1 != c_2;
	bool test_4 = c_2 != c_3;
	cout << "the result of \" " << c_1 << "==" << c_2 << " \" equals to " << test_1 << endl;
	cout << "the result of \" " << c_2 << "==" << c_3 << " \" equals to " << test_2 << endl;
	cout << "the result of \" " << c_1 << "!=" << c_2 << " \" equals to " << test_3 << endl;
	cout << "the result of \" " << c_2 << "!=" << c_3 << " \" equals to " << test_4 << endl;
	cout << "---------------------------------------------------------------------------" << endl;
}

int main()
{
	cout << "---------------------------------------------------------------------------" << endl;
	//测试<<运算符与>>运算符的重载
	TestIO();
	//测试*运算符的重载
	TestMultiply();
	//测试==与!=运算符的重载
	TestEqual();
}
```
