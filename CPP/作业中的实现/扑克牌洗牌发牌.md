#include <iostream>
#include <ctime>
using namespace std;

//一张牌
class Card
{
	//友元类
	friend class DeckOfCards;
	friend class FiveCardPoker;

private:
	//牌值
	int face;
	//花色
	int suit;

public:
	//默认构造函数
	Card() :face(0), suit(0) {}

	//牌值数组
	static string faces[];
	//花色数组
	static string suits[];

	//返回牌号
	string toSring()
	{
		//区分有花色的普通牌和大小王
		if (face == 13 || face == 14)
		{
			return faces[face] + "" + " of " + "" + "NONE";
		}
		else
		{
			return faces[face] + "" + " of " + "" + suits[suit];
		}
	}
};

string Card::faces[] = { "3","4","5","6","7","8","9","X","J","Q","K","A","2","BJ","RJ" };
string Card::suits[] = { "Spades","Hearts","Diamonds","Clubs" };

//一副完整的54张牌
class DeckOfCards
{
	//友元类
	friend class FiveCardPoker;

private:
	int currentCard;

public:
	//一副牌54张
	Card deck[54];

	//初始化牌组
	DeckOfCards()
	{
		Card temp;
		for (int i = 0; i < 13; i++)
		{
			for (int j = 0; j < 4; j++)
			{
				temp.face = i;
				temp.suit = j;
				deck[4 * i + j] = temp;
			}
		}
		Card BJ;
		BJ.face = 13;
		Card RJ;
		RJ.face = 14;
		deck[52] = BJ;
		deck[53] = RJ;
		//默认发的牌从第一张开始
		currentCard = 0;
	}

	//洗牌的实现
	void shuffle()
	{
		int random;
		srand((unsigned int)time(NULL));
		for (int k = 0; k < 54; k++)
		{
			//生成随机牌
			random = rand() % 54;
			do
			{
				random = rand() % 54;
			} while (random == k);
			Card tempo = deck[k];
			deck[k] = deck[random];
			deck[random] = tempo;
		}
	}

	//发牌
	Card dealCard()
	{
		return deck[currentCard];
		currentCard++;
	}

	//判断是否还有剩余的牌
	bool moreCard()
	{
		if (currentCard = 53)
		{
			return false;
		}
		else
		{
			return true;
		}
	}

	//展示牌组
	void showDeck()
	{
		cout << "Current Cards Deck Series: " << endl;
		for (int i = 0; i < 13; i++)
		{
			for (int j = 0; j < 4; j++)
			{
				cout << deck[4 * i + j].toSring() << "\t  ";
			}
			cout << endl;
		}
		cout << deck[52].toSring() << "\t  " << deck[53].toSring() << endl << endl;
	}
};

//一组5张扑克牌
class FiveCardPoker
{
private:
	Card deck_five[5];
	DeckOfCards carddeck;
	int count_face[15];
	int count_suit[4];

public:
	FiveCardPoker()
	{
		//洗牌
		carddeck.shuffle();
		//初始化数组元素为0
		for (int i = 0; i < 15; i++) {
			count_face[i] = 0;
		}
		for (int i = 0; i < 4; i++) {
			count_suit[i] = 0;
		}
		//发五张牌
		for (int i = 0; i < 5; i++)
		{
			//发牌
			deck_five[i] = carddeck.deck[i];
			//记录发到手里的牌的face类型及其出现个数，记录到数组内
			for (int j = 0; j < 15; j++)
			{
				if (deck_five[i].face == j)
				{
					count_face[j]++;
				}
			}
			//记录发到手里的牌的suit类型及其出现个数，记录到数组内
			for (int k = 0; k < 4; k++)
			{
				if (deck_five[i].face == k)
				{
					count_suit[k]++;
				}
			}
		}
		//冒泡排序手牌
		for (int k = 0; k < 5; k++)
		{
			for (int m = 0; m < 5; m++)
			{
				if (deck_five[k].face < deck_five[m].face)
				{
					int temp;
					temp = deck_five[k].face;
					deck_five[k].face = deck_five[m].face;
					deck_five[m].face = temp;
				}
			}
		}
	}

	//遍历展示手牌
	void showHandCards()
	{
		cout << "Your Hand Cards: " << endl;
		for (int i = 0; i < 5; i++)
		{
			cout << deck_five[i].toSring() << "   ";
		}
		cout << endl;
	}

	//检测手牌是否有对子
	bool testPairs()
	{
		int numPairs = 0;
		for (int i = 0; i < 15; i++)
		{
			if (count_face[i] == 2)
			{
				numPairs++;
			}
		}
		if (numPairs > 0)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}

	//检测手牌是否有两个对子
	bool testTwoPairs()
	{
		int numPairs = 0;
		for (int i = 0; i < 15; i++)
		{
			if (count_face[i] == 2)
			{
				numPairs++;
			}
		}
		if (numPairs == 2)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}

	//检测手牌是否有three of a kind
	bool testTriples()
	{
		int numTri = 0;
		for (int i = 0; i < 15; i++)
		{
			if (count_face[i] == 3)
			{
				numTri++;
			}
		}
		if (numTri > 0)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}

	//检测手牌是否有four of a kind
	bool testFour()
	{
		int numF = 0;
		for (int i = 0; i < 15; i++)
		{
			if (count_face[i] == 4)
			{
				numF++;
			}
		}
		if (numF > 0)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}

	//检测手牌是否都是同花色
	bool testSameSuit()
	{
		int num = 0;
		for (int i = 0; i < 4; i++)
		{
			if (count_suit[i] == 5)
			{
				num++;
			}
		}
		if (num > 0)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}

	//检测是否有顺子
	bool testStraight()
	{
		int getface[5];
		int between[4];
		int num = 0;
		//记录手排的face编号
		for (int i = 0; i < 5; i++)
		{
			getface[i] = deck_five[i].face;
		}
		//比较编号间的差值
		for (int j = 0; j < 4; j++)
		{
			between[j] = getface[j + 1] - getface[j];
		}
		//检查差值是否都为1
		for (int k = 0; k < 4; k++)
		{
			if (between[k] == 1)
			{
				num++;
			}
		}
		if (num == 4)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}

};

void testHandCards()
{
	FiveCardPoker handcard;

	cout << "whether has pairs:             " << handcard.testPairs() << endl;
	cout << "whether has two pairs:         " << handcard.testTwoPairs() << endl;
	cout << "whether has three of a kind:   " << handcard.testTriples() << endl;
	cout << "whether has four of a kind:    " << handcard.testFour() << endl;
	cout << "whether has five same suit:    " << handcard.testSameSuit() << endl;
	cout << "whether has straight:          " << handcard.testStraight() << endl;
}

int main()
{
	testShuffle();
	testHandCards();
}
